call plug#begin()

Plug 'jiangmiao/auto-pairs'
Plug 'tpope/vim-commentary'
Plug 'joshdick/onedark.vim'
Plug 'vim-python/python-syntax'
Plug 'junegunn/seoul256.vim'
Plug 'itchyny/lightline.vim'

call plug#end()

colorscheme onedark
set number
syntax on

let g:python_highlight_all = 1

set noswapfile
set incsearch
set tabstop=4 softtabstop=4
set shiftwidth=4
set expandtab
set smartindent
set nowrap
set smartcase
set noerrorbells
set nobackup
set nowritebackup

cnoremap ii <esc>
vnoremap ii <esc>
inoremap ii <esc>

" 'oo' to make new line without entering insert mode
nmap oo o<Esc>k

" ctrl+j to run currently editing python file
autocmd FileType python map <buffer> <C-J> :w<CR>:exec '!python3' shellescape(@%, 1)<CR>
autocmd FileType python imap <buffer> <C-J> <esc>:w<CR>:exec '!python3' shellescape(@%, 1)<CR>

autocmd InsertEnter * norm zz

set guifont=JetBrains_Mono_Medium:h13

set laststatus=2
let g:lightline = {
       \ 'colorscheme': 'wombat',
       \ }

set clipboard=unnamed
